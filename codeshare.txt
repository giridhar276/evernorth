import csv
workset = set()
try:
    with open("employeeinfo.csv") as fobj:
        header = fobj.readline()  # skipping the first line as it is header
        reader = csv.reader(fobj)
        # processing
        for line in reader:
            workclass = line[1]
            #print(workclass)
            workset.add(workclass)
        # displaying output
        print("--- workclasses----")
        for item in workset:
            print(item)

except Exception as err:
    print(err)
